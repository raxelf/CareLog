Table Users {
  id SERIAL pk
  email VARCHAR
  password TEXT
  role VARCHAR
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table Profiles {
  id SERIAL pk
  name VARCHAR
  gender VARCHAR
  birthOfDate DATE
  address VARCHAR
  phoneNumber VARCHAR
  specialization VARCHAR
  status VARCHAR
  UserId INTEGER
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table Queues {
  id SERIAL pk
  status VARCHAR
  reason VARCHAR
  scheduledAt DATE
  DoctorId INTEGER
  PatientId INTEGER
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table Histories {
  id SERIAL pk
  MedicalRecordId INTEGER
  PrescriptionId INTEGER
  QueueId INTEGER
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table MedicalRecords {
  id SERIAL pk
  emrCode VARCHAR
  anamnesis VARCHAR
  diagnosis VARCHAR
  doctorNotes VARCHAR
  status VARCHAR
  DoctorId INTEGER
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table Prescriptions {
  id SERIAL pk
  prescriptionCode VARCHAR
  MedicalRecordId INTEGER
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table PrescriptionDetails {
  id SERIAL pk
  quantity INTEGER
  dosage VARCHAR
  PrescriptionId INTEGER
  MedicineId INTEGER
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}

Table Medicines {
  id SERIAL pk
  name VARCHAR
  unit VARCHAR
  createdAt TIMESTAMP
  updatedAt TIMESTAMP
}
